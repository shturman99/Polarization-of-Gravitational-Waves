\section{GOGOBERIDZE 2007}

We adopt the spatial Fourier pair
\begin{align}
R_{ij}(\br,\tau) &= \int \frac{d^3k}{(2\pi)^3} e^{i\kb \cdot \br} F_{ij}(\kb,\tau), &
F_{ij}(\kb,\tau) &= \int d^3r \, e^{-i\kb \cdot \br} R_{ij}(\br,\tau),
\end{align}
and the temporal Fourier pair
\begin{align}
f(\eta_k,\tau) &= \int \frac{d\omega}{2\pi} e^{-i\omega \tau} \tilde f(\eta_k,\omega), &
\tilde f(\eta_k,\omega) &= \int d\tau \, e^{i\omega \tau} f(\eta_k,\tau).
\end{align}

Starting with all the definitions from Eqs.~23--26 of \cite{2007Gogoberidze},
\begin{align}
R_{ij}(\br,\tau) &= \int \frac{d^3k}{(2\pi)^3} e^{i\kb \cdot \br} F_{ij}(\kb,\tau) \nonumber\\
&= \int \frac{d^3k}{(2\pi)^3} e^{i\kb \cdot \br} A(k) P_{ij}(\khat) f(\eta_k,\tau) \nonumber\\
&= \int \frac{d^3k}{(2\pi)^3} e^{i\kb \cdot \br} A(k) P_{ij}(\khat)
\int \frac{d\omega}{2\pi} e^{-i\omega \tau}\tf(\eta_k,\omega).
\end{align}

Here $A(k) = C_k \epsilon^{2/3}k^{-11/3} /4\pi$, $P_{ij}(\khat) = \left( \delta_{ij}- \frac{k_{i}k_{j}}{k^{2}} \right)$, $f(\eta_k, \tau)= \exp(-\frac{\pi}{4}\eta_k^2 \tau^2)$, and $ \eta_k = \frac{1}{\sqrt{2\pi}}\epsilon^{1/3}k^{2/3}$.

Equation~28 of \cite{2007Gogoberidze} gives
\begin{equation}
    R_{ijij} (\xb',\xb'+ \br,\tau)=  R_{ii}(\br,\tau)R_{jj}(\br,\tau)+ \frac{1}{3}R_{ij}(\br,\tau)R_{ij}(\br,\tau).
\end{equation}

\begin{equation}
    H_{ijkl}(\xb',\kb,\omega) = \frac{1}{(2\pi)^{4}}\int d \tau d^3 \bxi \, e^{ i\omega \tau } e^{-i\kb \cdot \bxi} R_{ijkl}(\bxi,\tau).
\end{equation}

\begin{equation}
\label{eq: main}
    H_{ijij}(\xb',\kb,\omega) = \frac{1}{(2\pi)^{4}}\int d \tau d^3 \br \, e^{ i\omega \tau } e^{-i\kb \cdot \br}
    \left(
    R_{ii}(\br,\tau)R_{jj}(\br,\tau)+ \frac{1}{3}R_{ij}(\br,\tau)R_{ij}(\br,\tau)
    \right ).
\end{equation}

We substitute the spatial transforms for each product explicitly:
\begin{align}
R_{ii}(\br,\tau)R_{jj}(\br,\tau)
&= \int \frac{d^3q}{(2\pi)^3} \frac{d^3p}{(2\pi)^3}
e^{i(\bq+\bp)\cdot \br} F_{ii}(\bq,\tau) F_{jj}(\bp,\tau), \label{eq: Rijij-term1} \\
R_{ij}(\br,\tau)R_{ij}(\br,\tau)
&= \int \frac{d^3q}{(2\pi)^3} \frac{d^3p}{(2\pi)^3}
e^{i(\bq+\bp)\cdot \br} F_{ij}(\bq,\tau) F_{ij}(\bp,\tau). \label{eq: Rijij-term2}
\end{align}
Inserting Eqs.~\eqref{eq: Rijij-term1}--\eqref{eq: Rijij-term2} into Eq.~\eqref{eq: main} yields
\begin{align}
H_{ijij}(\xb',\kb,\omega)
&= \frac{1}{(2\pi)^{4}}\int d \tau d^3 \br \, e^{ i\omega \tau } e^{-i\kb \cdot \br}
\int \frac{d^3q}{(2\pi)^3} \frac{d^3p}{(2\pi)^3}
e^{i(\bq+\bp)\cdot \br} \nonumber\\
&\quad \times \left[ F_{ii}(\bq,\tau) F_{jj}(\bp,\tau)
+ \frac{1}{3} F_{ij}(\bq,\tau) F_{ij}(\bp,\tau) \right]. \label{eq: H-before-r}
\end{align}
The $\br$ integral gives
\begin{equation}
\int d^3r \, e^{-i\kb\cdot\br} e^{i(\bq+\bp)\cdot\br}
 = (2\pi)^3 \delta^{(3)}(\kb-\bq-\bp),
\end{equation}
so integrating over $\bp$ and renaming $\bq \to \kb_1$, $\ub = \kb-\kb_1$ with $u = |\ub|$, we obtain
\begin{align}
H_{ijij}(\xb',\kb,\omega)
&= \frac{1}{(2\pi)^{7}}\int d \tau \, e^{ i\omega \tau }
\int d^3k_{1} \left[ F_{ii}(\kb_1,\tau) F_{jj}(\ub,\tau)
+ \frac{1}{3} F_{ij}(\kb_1,\tau) F_{ij}(\ub,\tau) \right]. \label{eq: H-after-r}
\end{align}

Next insert the temporal Fourier representation
\begin{equation}
F_{ab}(\kb,\tau) = \int \frac{d\omega}{2\pi} e^{-i\omega \tau} \tilde F_{ab}(\kb,\omega),
\end{equation}
which gives
\begin{align}
H_{ijij}(\xb',\kb,\omega)
&= \frac{1}{(2\pi)^{7}}\int d^3k_{1}
\int d\tau \, e^{ i\omega \tau }
\int \frac{d\omega_{1}}{2\pi}\frac{d\omega_{2}}{2\pi}
e^{-i(\omega_{1}+\omega_{2})\tau} \nonumber\\
&\quad \times \left[ \tilde F_{ii}(\kb_1,\omega_{1}) \tilde F_{jj}(\ub,\omega_{2})
+ \frac{1}{3} \tilde F_{ij}(\kb_1,\omega_{1}) \tilde F_{ij}(\ub,\omega_{2}) \right]. \label{eq: H-before-omega}
\end{align}
The $\tau$ integral is
\begin{equation}
\int d\tau \, e^{i\omega \tau} e^{-i(\omega_{1}+\omega_{2})\tau}
= 2\pi \delta(\omega-\omega_{1}-\omega_{2}),
\end{equation}
so integrating over $\omega_{2}$ with $\omega_{2}=\omega-\omega_{1}$ yields the convolution form
\begin{align}
H_{ijij}(\xb',\kb,\omega)
&= \frac{1}{(2\pi)^{8}}\int d^3k_{1} \int d\omega_{1}
\left[ \tilde F_{ii}(\kb_1,\omega_{1}) \tilde F_{jj}(\ub,\omega-\omega_{1})
+ \frac{1}{3} \tilde F_{ij}(\kb_1,\omega_{1}) \tilde F_{ij}(\ub,\omega-\omega_{1}) \right]. \label{eq: H-convolution}
\end{align}

Now substitute the isotropic incompressible model
\begin{equation}
F_{ij}(\kb,\tau) = A(k) P_{ij}(\khat) f(\eta_k,\tau),
\qquad
\tilde F_{ij}(\kb,\omega) = A(k) P_{ij}(\khat) \tilde f(\eta_k,\omega),
\end{equation}
and evaluate the index contractions explicitly. First,
\begin{align}
P_{ii}(\khat) &= \delta_{ii} - \khat_i \khat_i
= 3 - 1 = 2. \label{eq: Pii}
\end{align}
Second,
\begin{align}
P_{ij}(\khat_1) P_{ij}(\hat{u})
&= (\delta_{ij}-\khat_{1i}\khat_{1j})(\delta_{ij}-\hat{u}_i \hat{u}_j) \nonumber\\
&= \delta_{ij}\delta_{ij} - \delta_{ij}\hat{u}_i\hat{u}_j - \khat_{1i}\khat_{1j}\delta_{ij}
+ \khat_{1i}\khat_{1j}\hat{u}_i\hat{u}_j \nonumber\\
&= 3 - 1 - 1 + (\khat_1\cdot\hat{u})^2
= 1 + (\khat_1\cdot\hat{u})^2. \label{eq: PijPij}
\end{align}
Using Eqs.~\eqref{eq: Pii}--\eqref{eq: PijPij} in Eq.~\eqref{eq: H-convolution} gives
\begin{align}
H_{ijij}(\xb',\kb,\omega)
&= \frac{1}{(2\pi)^{8}}\int d^3k_{1} \int d\omega_{1}
A(k_{1})A(u) \tilde f(\eta_{k_{1}},\omega_{1})\tilde f(\eta_{u},\omega-\omega_{1}) \nonumber\\
&\quad \times \left[ 4 + \frac{1}{3}\left(1+(\khat_{1}\cdot\hat{u})^2\right) \right]. \label{eq: H-with-P}
\end{align}
This is form of the Fourier  transform valid for all values of $k$ in, from now we will treat $\kb=0$ and $\kb\ne0$ separately

\section{Making Decaying Spectrum Dimensionless Murman + Titus}
Our integral begins as

\begin{equation}
    H_{ijij}(\bm{k},\omega) =
    \frac{1}{6}\int d^3k_1d\omega_1 g(\bm{k_1},w_1)g(\bm{k}-\bm{k_1},\omega -\omega_1) \mathfrak{K}
\end{equation}

where 

\begin{equation}
    \mathfrak{K}=\left[ 27-\frac{k^2}{k_1^2}+\frac{k^4}{2k^2_1u^2}+\frac{k_1^2}{2u^2}-\frac{k^2}{u^2}+\frac{u^2}{2k^2_1} \right] 
\end{equation}

with $u=|\bm{k}-\bm{k_1}|$ and 

\begin{equation}
    g(\bm{k},\omega) = \frac{\text{w}v_1^2}{8\pi k^2} \tau_1^2e^{i\omega\tau_1}(-i\omega\tau_1)^{-\frac{5}{3}}\Gamma\left(\frac{1}{3},-i\omega\tau_1\right)
\end{equation}

Let's begin by separating the $k$ dependence from the $\omega$ dependence. 
\begin{equation}
    g(\bm{k},\omega) = \frac{\text{w}v_1^2}{8\pi k^2} \tau_1^2 g(\omega)
\end{equation}
 where 
 \begin{equation}
     g(\omega)=e^{i\omega\tau_1}(-i\omega\tau_1)^{-\frac{5}{3}}\Gamma\left(\frac{1}{3},-i\omega\tau_1\right)
 \end{equation}

If we substitute $q=\omega \tau_1$ (and likewise $q_1=\omega_1 \tau_1$ and $dq=\tau_1d\omega$) then 

 \begin{equation}
     g(q)=e^{iq}(-iq)^{-\frac{5}{3}}\Gamma\left(\frac{1}{3},-iq\right)
 \end{equation}

Using these substitutions:

\begin{equation}
    H_{ijij}(\bm{k},q) =
    \frac{1}{6}\int d^3k_1dq_1 \frac{1}{\tau_1}\frac{\text{w}v_1^2\tau_1^2}{8\pi k^2_1} \frac{\text{w}v_1^2\tau_1^2}{8\pi u^2} g(q_1)g(q -q_1) \mathfrak{K}
\end{equation}

Letting $A=\frac{\text{w}^2v_1^4\tau_1^3}{384\pi^2}$
\begin{equation}
    H_{ijij}(\bm{k},q) =
    A \int^\infty_0dk_1 \int^{2\pi}_0d\phi_1 \int^\pi_0d\theta_1 k_1^2\sin{\theta_1} \int_0^\infty dq_1 
     \frac{1}{k_1^2u^2}\cdot\mathfrak{K}
     \cdot g(q_1)g(q-q_1)
\end{equation}


Because we are assuming isotropic turbulence,  we can set the $\theta_1=0$ axis with to align with $\bm{k}$. When we do, we find that the integral over $\phi_1$ is simply $2\pi$. We also can relate $u$ to $k_1$, $\theta_1$ and $k$ using the cosine law.

\begin{equation}
    u = \sqrt{k_1^2+k^2-2k_1k \cos{\theta_1}}
\end{equation}

Therefore 
 \begin{equation}
    du = \frac{k_1k}{u}\sin{\theta_1}d\theta_1
\end{equation}

and the bounds are $\theta_1=0 \rightarrow |k_1-k|$, and $\theta_1=\pi \rightarrow k_1+k$.

\begin{equation}
    H_{ijij}(k,q) =
    2\pi A \int^\infty_0dk_1  \int^{k+k_1}_{|k-k_1|}du \frac{\mathfrak{K}}{kk_1u}\int_0^\infty dq_1 
      g(q_1)g(q-q_1)
\end{equation}

At this point, all of the integrals except the integral over $q$ (which is dimensionless) can be analyzed analytically. However, we will go ahead and make the integral fully dimensionless. We also wish to ensure that we integrate only over the correct inertial range from $k_0$ to $k_d$. We will define the following dimensionless variables: 
\begin{equation}
    R^\frac{3}{4}=\frac{k_d}{k_0}, \ \frac{u}{k_0}=y, \ \frac{k_1}{k_0} = z, \ \frac{k}{k_0} = p, \ \frac{\omega}{k_0}=\omega \tau_1 =q
\end{equation}

Using these new variables

\begin{equation}
    \mathfrak{K}=\left[ 
    27 -\frac{z^2}{p^2}+ \frac{p^4}{2z^2y^2} + 
    \frac{z^2}{2y^2} 
    - \frac{p^2}{y^2} + \frac{y^2}{2p^2}
    \right]
\end{equation}

\begin{equation}
    dk_1 = k_0dz,\  du = k_0dy
\end{equation}

We also need to find the bounds for the y integral.

\begin{equation}
    \max(k_1+k,k_d) \rightarrow \max(R^\frac{3}{4},p+z), \ \min(k_0,\bm{k_1}+\bm{k}) \rightarrow \min(1,|p-z|) 
\end{equation}

Finally we find:

\begin{equation}
