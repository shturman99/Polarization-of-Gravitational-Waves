\subsection{Polarization}
For the purposes of this work, we have the following metric.
\begin{equation}
    ds^{2}=-dt^{2}+a(t)(\delta_{ij}+h_{ij}(t,\vec{x}))dx^{i}dx^{j}
\end{equation}

The metric perturbations can be expressed as plane waves:

\begin{equation} h_{ij}(t, \vec{x}) = \sum_{\lambda} \int_{-\infty}^{\infty} df \int d^2 \n
e^{2i\pi f(t - \n \cdot \vec{x})} e^{\lambda}_{ij}(\n) h_{\lambda}(f, \n) \end{equation}

where $\lambda = {+,\times}$ represents the two polarization states of the gravitational wave. These modes clearly depend only on the direction $\n$ of the gravitational wave, but there is also a dependence on the parameters frequency of the modes. This frequency dependence, as assumed in parameters\cite{2023AnilKumar}, applies to an isotropic stochastic gravitational wave background. 

\begin{equation}
    \langle h_{\lambda}(f,\hat{n})h_{\lambda'}^{*}(f',\hat{n}) \rangle =\delta (f-f')\delta(\hat{n}-\hat{n}')\mathcal{P}_{\lambda\lambda'}(f)
    % maek sure that normalization is correct $4\pi$
\end{equation}
\begin{equation*}
\left\langle\tilde{h}_\lambda(f, \hat{n}) \tilde{h}_{\lambda^{\prime}}^*(f^{\prime}, \hat{n}^{\prime})\right\rangle=\frac{1}{4} \mathcal{P}(f, \hat{n}) \delta\left(f-f^{\prime}\right) \delta_{\lambda \lambda^{\prime}} \delta^2(\hat{n}, \hat{n}^{\prime}),    
\end{equation*}

The $\mathcal{P}_{\lambda\lambda'}$ is a spectra density and its dependence on the polarization states of the gravitational wave can be expressed using Stokes's parameters:
\begin{equation}
\mathcal{P}_{p, p^{\prime}}(f, \hat{\Omega})=\left(
\begin{array}{cc}
I(f, \hat{\Omega})+Q(f, \hat{\Omega}) & U(f, \hat{\Omega})-i V(f, \hat{\Omega}) \\
U(f, \hat{\Omega})+i V(f, \hat{\Omega}) & I(f, \hat{\Omega})-Q(f, \hat{\Omega})
\end{array}\right)
\end{equation}
with the following definitions of each parameter:
\begin{equation}
\begin{aligned}
& I(f, \hat{n})=\frac{1}{2}\left\langle | \tilde{h}_{+}|^2+|\tilde{h}_{\times}|^2\right\rangle \\
& Q(f, \hat{n})=\frac{1}{2}\left\langle | \tilde{h}_{+}|^2-|\tilde{h}_{\times}|^2\right\rangle \\
& U(f, \hat{n})=\operatorname{Re}\left\langle\tilde{h}_{+}^* \tilde{h}_{\times}\right\rangle=\frac{1}{2}\left\langle\tilde{h}_{+}^* \tilde{h}_{\times}+\tilde{h}_{\times}^* \tilde{h}_{+}\right\rangle \\
& V(f, \hat{n})=\operatorname{Im}\left\langle\tilde{h}_{+}^* \tilde{h}_{\times}\right\rangle=\frac{1}{2 i}\left\langle\tilde{h}_{+}^* \tilde{h}_{\times}-\tilde{h}_{\times}^* \tilde{h}_{+}\right\rangle
\end{aligned}
\end{equation}
The usefulness of this decomposition is the explicit transformations of this quantities under rotations:
\begin{equation}
\begin{gathered}
I(\hat{n})=I_0 \sum_{L=0}^{\infty} \sum_{M=-L}^L c_{L M}^I Y_{L M}(\hat{n}), \\
V(\hat{n})=I_0 \sum_{L=0}^{\infty} \sum_{M=-L}^L c_{L M}^V Y_{L M}(\hat{n}), \\
P_{ \pm}(\hat{n})=I_0 \sum_{L=4}^{\infty} \sum_{M=-L}^L c_{L M \pm 4}^{ \pm} Y_{L M}(\hat{n}),
\end{gathered}
\end{equation}
where $P_{\pm}=Q\pm iU$. Here we see that $I$ and $V$ transform like scalars while $P_{\pm}$ transforms like spin-4 file.\\
Conversion from the linear polarization basis ($+$, $\times$) typically used in gravitational wave (GW) analyses, to the helical polarization basis, which consists of left-handed (L) and right-handed (R) circular polarization states. This helical basis is particularly useful when analyzing parity-violating phenomena or when clearly isolating circular polarization components.

We start by defining the relationship between the linear polarization tensors and the helical polarization tensors as done in the appendix B of \cite{2019Domcke}. For a gravitational wave propagating in the direction $\hat{n}$, the polarization tensors can be transformed as follows:

\begin{align}
&e^{R}_{ij}(\hat{n}) = \frac{e^{+}_{ij}(\hat{n}) + i e^{\times}_{ij}(\hat{n})}{\sqrt{2}}, \\
&e^{L}_{ij}(\hat{n}) = \frac{e^{+}_{ij}(\hat{n}) - i e^{\times}_{ij}(\hat{n})}{\sqrt{2}}
\end{align}

Conversely, we express the linear polarization tensors in terms of the helical polarization tensors:

\begin{equation}
e^{+}_{ij}(\hat{n}) = \frac{e^{R}_{ij}(\hat{n}) + e^{L}_{ij}(\hat{n})}{\sqrt{2}}, \quad
e^{\times}_{ij}(\hat{n}) = \frac{e^{R}_{ij}(\hat{n}) - e^{L}_{ij}(\hat{n})}{\sqrt{2}i}
\end{equation}

The GW mode expansions in the helical basis become:

\begin{equation}
h_{ij}(t, \vec{x}) = \sum_{\lambda=R,L} \int_{-\infty}^{\infty} df \int d^2 \hat{n}
e^{2\pi i f(t - \hat{n} \cdot \vec{x})} e^{\lambda}{ij}(\hat{n}) h_{\lambda}(f, \hat{n})
\end{equation}

The power spectral densities in the helical basis can be expressed analogously to the linear case, with the Stokes parameters now relating directly to the helical amplitudes.

\begin{equation}
\begin{aligned}
&I(f, \hat{n}) = \frac{1}{2}\left\langle |\tilde{h}_{R}|^2 + |\tilde{h}_{L}|^2 \right\rangle, \\
&V(f, \hat{n}) = \frac{1}{2}\left\langle |\tilde{h}_{R}|^2 - |\tilde{h}_{L}|^2 \right\rangle, \\
&Q(f, \hat{n}) + iU(f, \hat{n}) = \left\langle \tilde{h}_{R}\tilde{h}_{L}^* \right\rangle.
\end{aligned}
\end{equation}

Here, $I$ and $V$ are directly interpreted as the total intensity and the circular polarization, respectively, while $Q$ and $U$ measure linear polarization correlations within the helical framework. This decomposition facilitates explicit studies of GW polarization properties, particularly under rotations and boosts, relevant for detecting and characterizing anisotropic or polarized stochastic gravitational wave backgrounds.
